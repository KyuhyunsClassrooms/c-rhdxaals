## 3. 데이터 구조 설계
(프로그램에 필요한 핵심 변수, 배열을 설계하세요. 필요하면 구조체(struct)를 사용할 수도 있습니다.)

// 상품 정보를 구조체로 묶어서 관리
struct Product {
    char name[30];     // 상품 이름
    int price;         // 상품 가격
    int quantity;      // 상품 수량
};

// 상품 10개를 저장할 구조체 배열
struct Product products[10];
int count = 0;         // 현재 등록된 상품 수
